'use strict';

                /**
                * Plumbing to define constants
                * @param name
                * @param value
                */
                function define(name, value) {
                Object.defineProperty(exports, name, {
                value: value,
                enumerable: true
                });
                }

                define("parseErrorMsg", "Could not decode request: JSON parsing failed");
