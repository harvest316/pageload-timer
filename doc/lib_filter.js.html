/*jslint node: true */
'use strict';
var logger = require('../utils/logger');
var constants = require('../lib/constants'); // for the required JSON parsing error message
var async = require('async'); // enables faster responses for large payloads
var request = require('./request.js');

/**
 * The core of the web service is this method which calculates pageload times for the given URLs
 */
var pageLoadTimer = (function () {
    return {

        /**
         * Returns Page Load Times for a Given List of URLs
         * @param req
         * @param callback
         */
        get: function (req, callback) {
            var pageLoadTimer = [];
            //TODO: Clean up naming
            if (request.isValid(req)) {
                // asynchronously process payload
                async.forEach(req.payload, function (show, callback) {
                    // load the page and calculate time
                    pageLoadTimer.push({image: show.image.showImage, slug: show.slug, title: show.title});
                    callback();
                }, function (err) { // final
                    if (err) { // treat all exceptions as parsing error
                        logger.error((new Date()).toUTCString() + ' Error in get loop:', err.message);
                        logger.error(err.stack);
                        callback(constants.parseErrorMsg, null); // the required error message
                    } else {
                        callback(null, pageLoadTimer); // success
                    }
                });
            } else {
                callback(constants.parseErrorMsg, null); // the required error message
            }

        }
    };

}());

exports = module.exports = pageLoadTimer;
